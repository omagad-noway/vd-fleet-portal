function doGet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheets = ss.getSheets(); 
  let allData = [];
  
  sheets.forEach(sheet => {
    const sheetName = sheet.getName();
    const range = sheet.getDataRange();
    const data = range.getValues();
    
    if (data.length < 2) return; 
    
    const expectedHeaders = ["DATE", "TRUCK", "STATEORIGIN", "STATEDESTINATION", "DEADHEAD", "LOADEDMILES", "GROSS", "TYPE"];
    const actualHeaders = data[0].map(h => h.toString().replace(/[:\s]/g, '').toUpperCase().trim());
    
    const sheetData = data.slice(1).filter(row => row[0] !== "").map((row) => {
      let obj = {};
      expectedHeaders.forEach(req => {
        const colIndex = actualHeaders.indexOf(req);
        if (colIndex > -1) {
          obj[req] = row[colIndex];
        } else {
          obj[req] = ""; 
        }
      });
      obj["_SOURCE_SHEET"] = sheetName;
      return obj;
    });
    
    allData = allData.concat(sheetData);
  });
  
  return ContentService.createTextOutput(JSON.stringify(allData))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  try {
    const val = JSON.parse(e.postData.contents);
    
    const getTargetSheet = (dateStr) => {
      const dateObj = new Date(dateStr);
      const options = { month: 'long', year: 'numeric' };
      const tabName = new Intl.DateTimeFormat('en-US', { ...options, timeZone: 'UTC' }).format(dateObj);
      
      let sheet = ss.getSheetByName(tabName);
      if (!sheet) {
        sheet = ss.insertSheet(tabName);
        sheet.appendRow(["DATE", "TRUCK", "STATE ORIGIN", "STATE DESTINATION", "DEADHEAD", "LOADED MILES", "GROSS", "TYPE"]);
        sheet.getRange("A1:H1").setFontWeight("bold").setBackground("#f3f3f3");
      }
      return sheet;
    };

    const toDateKey = (d) => {
      if (!d) return "";
      const dateObj = new Date(d);
      return dateObj.getUTCFullYear() + "-" + (dateObj.getUTCMonth() + 1) + "-" + dateObj.getUTCDate();
    };

    // MULTI-LEVEL SORT: Date then Truck
    const autoSortSheet = (sheet) => {
      const lastRow = sheet.getLastRow();
      if (lastRow > 1) {
        sheet.getRange(2, 1, lastRow - 1, 8).sort([
          {column: 1, ascending: true}, 
          {column: 2, ascending: true}
        ]);
      }
    };

    if (val.action === "add") {
      const targetSheet = getTargetSheet(val.DATE);
      targetSheet.appendRow([
        val.DATE, val.TRUCK, val.STATEORIGIN, val.STATEDESTINATION, 
        val.DEADHEAD, val.LOADEDMILES, val.GROSS, val.TYPE || "FULL"
      ]);
      
      autoSortSheet(targetSheet); 
      return ContentService.createTextOutput("Added and Sorted").setMimeType(ContentService.MimeType.TEXT);

    } else if (val.action === "edit") {
      const oldSheet = getTargetSheet(val.oldDate);
      const data = oldSheet.getDataRange().getValues();
      let updated = false;

      for (let i = 1; i < data.length; i++) {
        if (toDateKey(data[i][0]) === toDateKey(val.oldDate) && 
            data[i][1].toString().trim() === val.oldTruck.toString().trim() &&
            data[i][6].toString() === val.oldGross.toString()) {
          
          const newTabName = new Intl.DateTimeFormat('en-US', { month: 'long', year: 'numeric', timeZone: 'UTC' }).format(new Date(val.DATE));
          
          if (oldSheet.getName() !== newTabName) {
            oldSheet.deleteRow(i + 1);
            const newSheet = getTargetSheet(val.DATE);
            newSheet.appendRow([val.DATE, val.TRUCK, val.STATEORIGIN, val.STATEDESTINATION, val.DEADHEAD, val.LOADEDMILES, val.GROSS, val.TYPE]);
            autoSortSheet(newSheet);
          } else {
            oldSheet.getRange(i + 1, 1, 1, 8).setValues([[
              val.DATE, val.TRUCK, val.STATEORIGIN, val.STATEDESTINATION, 
              val.DEADHEAD, val.LOADEDMILES, val.GROSS, val.TYPE || "FULL"
            ]]);
            autoSortSheet(oldSheet);
          }
          updated = true;
          break;
        }
      }
      return ContentService.createTextOutput("Updated").setMimeType(ContentService.MimeType.TEXT);

    } else if (val.action === "delete") {
      const targetSheet = getTargetSheet(val.date);
      const data = targetSheet.getDataRange().getValues();
      let found = false;
      for (let i = data.length - 1; i >= 1; i--) {
        if (toDateKey(data[i][0]) === toDateKey(val.date) && 
            data[i][1].toString().trim() === val.truck.toString().trim() && 
            data[i][6].toString() === val.gross.toString()) {
          targetSheet.deleteRow(i + 1);
          found = true;
          break;
        }
      }
      return ContentService.createTextOutput("Deleted").setMimeType(ContentService.MimeType.TEXT);
    }
  } catch (err) {
    return ContentService.createTextOutput("Error: " + err).setMimeType(ContentService.MimeType.TEXT);
  }
}
